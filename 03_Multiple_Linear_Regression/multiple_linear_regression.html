import numpy as np
import statsmodels.api as sm

def backward_elim(X, y, sl=0.05):
    # تحويل X إلى مصفوفة وتأكيد نوع البيانات ليكون فلوت
    X = np.array(X, dtype=float)
    # إضافة عمود المحايد (Ones) في البداية لحساب الـ Intercept (الحد الثابت b0)
    X = np.append(np.ones((len(X), 1)), X, axis=1)
    
    while True:
        # تدريب نموذج الانحدار الخطي
        m = sm.OLS(y, X).fit()
        p = m.pvalues
        
        # إذا كانت أكبر قيمة p-value تتخطى حد المعنوية (sl)
        if p.max() > sl:
            # نحدد مكان العمود صاحب أكبر p-value ونحذفه
            max_p_idx = np.argmax(p)
            X = np.delete(X, max_p_idx, axis=1)
        else: 
            # إذا كانت كل القيم أقل من الحد، توقف
            break
            
    return X

# تشغيل الدالة
X_opt = backward_elim(X, y)

import numpy as np
import pandas as pd
from statsmodels.stats.outliers_influence import variance_inflation_factor

# 1. Force the training data to be a clean, numeric numpy array (fixes the data type error)
if isinstance(X_train, pd.DataFrame):
    X_clean = X_train.to_numpy(dtype=float)
else:
    X_clean = np.array(X_train, dtype=float)

# 2. Create an empty DataFrame to store the VIF results
vif_df = pd.DataFrame()

# 3. Add a column for feature indices (or use feature names if available)
if isinstance(X_train, pd.DataFrame):
    vif_df['Feature'] = X_train.columns
else:
    vif_df['Feature'] = [f'Feature_{i}' for i in range(X_clean.shape[1])]
